    **1. Create `docker-compose.yml`** at project root (per INFR-02):

    ```yaml
    services:
      db:
        image: postgres:17-alpine
        environment:
          POSTGRES_DB: prosperity
          POSTGRES_USER: prosperity
          POSTGRES_PASSWORD: prosperity
        ports:
          - "5432:5432"
        volumes:
          - pgdata:/var/lib/postgresql/data
        healthcheck:
          test: ["CMD-SHELL", "pg_isready -U prosperity"]
          interval: 5s
          timeout: 3s
          retries: 5

      backend:
        build:
          context: .
          dockerfile: Dockerfile.backend
        ports:
          - "8080:8080"
        environment:
          SPRING_DATASOURCE_URL: jdbc:postgresql://db:5432/prosperity
          SPRING_DATASOURCE_USERNAME: prosperity
          SPRING_DATASOURCE_PASSWORD: prosperity
        depends_on:
          db:
            condition: service_healthy
        healthcheck:
          test: ["CMD-SHELL", "curl -f http://localhost:8080/actuator/health || exit 1"]
          interval: 10s
          timeout: 5s
          retries: 3
          start_period: 30s

      caddy:
        image: caddy:2.11-alpine
        ports:
          - "80:80"
          - "443:443"
        volumes:
          - ./Caddyfile:/etc/caddy/Caddyfile
          - ./frontend/dist/frontend/browser:/srv
          - caddy_data:/data
          - caddy_config:/config
        depends_on:
          backend:
            condition: service_healthy

    volumes:
      pgdata:
      caddy_data:
      caddy_config:
    ```

    **2. Create `Dockerfile.backend`** at project root:

    Multi-stage build:
    ```dockerfile
    # Build stage
    FROM eclipse-temurin:21-jdk AS build
    WORKDIR /app
    COPY backend/.mvn .mvn
    COPY backend/mvnw backend/pom.xml ./
    RUN chmod +x mvnw && ./mvnw dependency:go-offline -B
    COPY backend/src src
    COPY backend/checkstyle.xml checkstyle.xml
    RUN ./mvnw package -DskipTests -B

    # Runtime stage
    FROM eclipse-temurin:21-jre
    WORKDIR /app
    COPY --from=build /app/target/*.jar app.jar
    EXPOSE 8080
    ENTRYPOINT ["java", "-jar", "app.jar"]
    ```

    **3. Create `Caddyfile`** at project root:

    ```
    :80 {
        handle /api/* {
            reverse_proxy backend:8080
        }

        handle /actuator/* {
            reverse_proxy backend:8080
        }

        handle {
            root * /srv
            try_files {path} /index.html
            file_server
        }
    }
    ```

    The `/actuator/*` route is needed for health checks from outside. The `try_files {path} /index.html` enables Angular SPA routing.
