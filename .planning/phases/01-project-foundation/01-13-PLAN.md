    Then run the pnpm commands.

    **2. Create `lefthook.yml`** at project root (per INFR-10):

    ```yaml
    pre-commit:
      parallel: true
      commands:
        java-format:
          glob: "*.java"
          run: cd backend && ./mvnw spotless:check -q
        java-lint:
          glob: "*.java"
          run: cd backend && ./mvnw checkstyle:check -q
        frontend-lint:
          root: "frontend/"
          glob: "*.{ts,html}"
          run: cd frontend && pnpm lint
        frontend-format:
          root: "frontend/"
          glob: "*.{ts,html,css,json}"
          run: cd frontend && pnpm format:check
    ```

    The `parallel: true` runs all 4 commands in parallel (Lefthook feature). Each command is scoped to its glob pattern so it only runs when relevant files are staged. Frontend hooks use `cd frontend &&` prefix to ensure the correct working directory for pnpm commands regardless of Lefthook's `root:` behavior.

    **3. Create `.github/workflows/ci.yml`** (per Roadmap success criteria 9):

    ```yaml
    name: CI

    on:
      push:
        branches: [main]
      pull_request:
        branches: [main]

    jobs:
      backend:
        runs-on: ubuntu-latest
        services:
          postgres:
            image: postgres:17-alpine
            env:
              POSTGRES_DB: prosperity
              POSTGRES_USER: prosperity
              POSTGRES_PASSWORD: prosperity
            ports:
              - 5432:5432
            options: >-
              --health-cmd "pg_isready -U prosperity"
              --health-interval 5s
              --health-timeout 3s
              --health-retries 5
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-java@v4
            with:
              distribution: temurin
              java-version: 21
              cache: maven
          - name: Build and verify (compile + test + checkstyle + spotless + jacoco + owasp)
            run: cd backend && ./mvnw verify -B
            env:
              SPRING_DATASOURCE_URL: jdbc:postgresql://localhost:5432/prosperity
              SPRING_DATASOURCE_USERNAME: prosperity
              SPRING_DATASOURCE_PASSWORD: prosperity
          - name: Cache OWASP NVD data
            uses: actions/cache@v4
            with:
              path: ~/.dependency-check/
              key: owasp-nvd-${{ hashFiles('backend/pom.xml') }}
              restore-keys: owasp-nvd-
          - name: Upload JaCoCo coverage report
            uses: actions/upload-artifact@v4
            if: always()
            with:
              name: jacoco-report
              path: backend/target/site/jacoco/

      frontend:
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - uses: actions/setup-node@v4
            with:
              node-version: 22
          - uses: pnpm/action-setup@v4
          - name: Install dependencies
            run: cd frontend && pnpm install --frozen-lockfile
          - name: Lint
            run: cd frontend && pnpm lint
          - name: Format check
            run: cd frontend && pnpm format:check
          - name: Build
            run: cd frontend && pnpm build
    ```

    The CI has two parallel jobs: `backend` and `frontend`. The backend job includes a PostgreSQL service container so Flyway migrations can run during `./mvnw verify`. The OWASP NVD database is cached to avoid slow first runs in CI. JaCoCo report is uploaded as artifact for visibility (per D-08).
