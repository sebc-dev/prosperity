```java
// Arrange: create admin via setup
mockMvc.perform(post("/api/auth/setup")
    .with(csrf())
    .contentType(MediaType.APPLICATION_JSON)
    .content("""
        {"email":"admin@test.com","password":"SecurePass123!","displayName":"Admin"}
        """))
    .andExpect(status().isCreated());

// Act: login
var result = mockMvc.perform(post("/api/auth/login")
    .with(csrf())
    .contentType(MediaType.APPLICATION_JSON)
    .content("""
        {"email":"admin@test.com","password":"SecurePass123!"}
        """));

// Assert
result.andExpect(status().isOk())
    .andExpect(jsonPath("$.displayName").value("Admin"));
```
