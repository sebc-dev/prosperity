**header.ts** — Header component with app title and logout button:

```typescript
import { Component, inject } from '@angular/core';
import { Router } from '@angular/router';
import { AuthService } from '../auth/auth.service';
import { ButtonModule } from 'primeng/button';
import { MenubarModule } from 'primeng/menubar';

@Component({
  selector: 'app-header',
  standalone: true,
  imports: [ButtonModule, MenubarModule],
  template: `
    <header class="h-16 flex items-center justify-between px-6 bg-surface-0 border-b border-surface-200">
      <span class="text-lg font-semibold">Prosperity</span>
      <p-button
        label="Deconnexion"
        [text]="true"
        severity="primary"
        (onClick)="onLogout()"
        [loading]="loggingOut"
      />
    </header>
  `,
})
export class Header {
  private authService = inject(AuthService);
  private router = inject(Router);
  loggingOut = false;

  onLogout(): void {
    this.loggingOut = true;
    this.authService.logout().subscribe({
      next: () => this.router.navigate(['/login']),
      error: () => {
        this.loggingOut = false;
        this.router.navigate(['/login']);
      },
    });
  }
}
```

Per D-10: minimal header with app title left, logout button right. Per UI-SPEC: `h-16`, white background, bottom border.

**app.routes.ts** — Complete route configuration:

```typescript
import { Routes } from '@angular/router';
import { authGuard, unauthenticatedGuard, noAdminGuard } from './auth/auth.guard';

export const routes: Routes = [
  {
    path: 'setup',
    loadComponent: () => import('./auth/setup').then(m => m.Setup),
    canActivate: [noAdminGuard],
  },
  {
    path: 'login',
    loadComponent: () => import('./auth/login').then(m => m.Login),
    canActivate: [unauthenticatedGuard],
  },
  {
    path: '',
    loadComponent: () => import('./layout/layout').then(m => m.Layout),
    canActivate: [authGuard],
    children: [
      {
        path: 'dashboard',
        loadComponent: () => import('./dashboard/dashboard').then(m => m.Dashboard),
      },
      {
        path: '',
        redirectTo: 'dashboard',
        pathMatch: 'full',
      },
    ],
  },
  {
    path: '**',
    redirectTo: '',
  },
];
```

Route logic per UI-SPEC routing table:
- `/setup` → Setup component, `noAdminGuard` redirects to `/login` if admin exists
- `/login` → Login component, `unauthenticatedGuard` redirects to `/dashboard` if already authenticated
- `/` → Layout shell (authGuard), children include `/dashboard` and default redirect
- `**` → Redirects to `/` (which redirects to `/dashboard`, guard chain handles auth)

Uses lazy loading (`loadComponent`) for all routes.
