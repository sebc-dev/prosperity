enum account_type: courant, livret, epargne, especes, credit
enum user_role: admin, member

table account_members
  col account_id: UUID NOT NULL
  col default_share_ratio: NUMERIC(5, 4) NOT NULL
  col id: UUID NOT NULL
  col joined_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col user_id: UUID NOT NULL
  pk pk_account_members(id)
  fk fk_account_members_account_id_accounts: (account_id) -> accounts(id) ON DELETE CASCADE
  fk fk_account_members_user_id_users: (user_id) -> users(id) ON DELETE RESTRICT
  idx ix_account_members_user_id INDEX: user_id
  idx uq_account_members_account_id_user_id UNIQUE: account_id, user_id

table accounts
  col archived_at: TIMESTAMP WITH TIME ZONE NULL
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col currency: VARCHAR(3) NOT NULL
  col household_id: UUID NOT NULL
  col id: UUID NOT NULL
  col name: VARCHAR(120) NOT NULL
  col owner_id: UUID NULL
  col type: account_type NOT NULL
  pk pk_accounts(id)
  fk fk_accounts_household_id_household: (household_id) -> household(id)
  fk fk_accounts_owner_id_users: (owner_id) -> users(id) ON DELETE RESTRICT
  idx ix_accounts_owner_id INDEX: owner_id

table admin_audit_logs

  col actor_user_id: UUID NULL
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col id: UUID NOT NULL
  col metadata: JSONB NULL
  col target_email: VARCHAR(254) NULL
  col target_user_id: UUID NULL
  pk pk_admin_audit_logs(id)
  fk fk_admin_audit_logs_actor_user_id_users: (actor_user_id) -> users(id) ON DELETE SET NULL
  fk fk_admin_audit_logs_target_user_id_users: (target_user_id) -> users(id) ON DELETE SET NULL
  idx ix_admin_audit_logs_actor_user_id_created_at INDEX: actor_user_id, created_at
  idx ix_admin_audit_logs_target_user_id INDEX: target_user_id

table alembic_version
  col version_num: VARCHAR(32) NOT NULL
  pk alembic_version_pkc(version_num)

table budget_contributors
  col budget_id: UUID NOT NULL
  col id: UUID NOT NULL
  col user_id: UUID NOT NULL
  pk pk_budget_contributors(id)
  fk fk_budget_contributors_budget_id_budgets: (budget_id) -> budgets(id) ON DELETE CASCADE
  fk fk_budget_contributors_user_id_users: (user_id) -> users(id) ON DELETE RESTRICT
  idx ix_budget_contributors_user_id INDEX: user_id
  idx uq_budget_contributors_budget_id_user_id UNIQUE: budget_id, user_id

table budget_threshold_alerts
  col budget_id: UUID NOT NULL
  col id: UUID NOT NULL
  col period_start: DATE NOT NULL
  col threshold_pct: SMALLINT NOT NULL
  pk pk_budget_threshold_alerts(id)
  fk fk_budget_threshold_alerts_budget_id_budgets: (budget_id) -> budgets(id) ON DELETE CASCADE
  idx uq_budget_threshold_alerts_dedup UNIQUE: budget_id, period_start, threshold_pct

table budgets
  col amount_cents: BIGINT NOT NULL
  col archived_at: TIMESTAMP WITH TIME ZONE NULL
  col carry_over_remainder: BOOLEAN NOT NULL
  col category_id: UUID NOT NULL
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col created_by: UUID NOT NULL
  col currency: VARCHAR(3) NOT NULL
  col id: UUID NOT NULL
  col period_kind: VARCHAR NOT NULL
  col period_start: DATE NOT NULL
  col scope: VARCHAR NOT NULL
  pk pk_budgets(id)
  fk fk_budgets_category_id_categories: (category_id) -> categories(id) ON DELETE RESTRICT
  fk fk_budgets_created_by_users: (created_by) -> users(id) ON DELETE RESTRICT
  idx ix_budgets_active INDEX: category_id) WHERE (archived_at IS NULL
  idx ix_budgets_category_id INDEX: category_id
  idx ix_budgets_created_by INDEX: created_by

table categories
  col archived_at: TIMESTAMP WITH TIME ZONE NULL
  col color: VARCHAR(7) NULL
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col icon: VARCHAR NULL
  col id: UUID NOT NULL
  col name: VARCHAR(120) NOT NULL
  col parent_id: UUID NULL
  pk pk_categories(id)
  fk fk_categories_parent_id_categories: (parent_id) -> categories(id) ON DELETE RESTRICT
  idx ix_categories_active INDEX: parent_id) WHERE (archived_at IS NULL
  idx ix_categories_parent_id INDEX: parent_id

table debts
  col account_id: UUID NOT NULL
  col amount_cents: BIGINT NOT NULL
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col currency: VARCHAR(3) NOT NULL
  col from_user_id: UUID NOT NULL
  col id: UUID NOT NULL
  col materialization_trace: VARCHAR NULL
  col origin: VARCHAR NOT NULL
  col share_ratio: NUMERIC(5, 4) NOT NULL
  col source_transaction_id: UUID NOT NULL
  col to_user_id: UUID NOT NULL
  pk pk_debts(id)
  fk fk_debts_account_id_accounts: (account_id) -> accounts(id) ON DELETE RESTRICT
  fk fk_debts_from_user_id_users: (from_user_id) -> users(id) ON DELETE RESTRICT
  fk fk_debts_source_transaction_id_transactions: (source_transaction_id) -> transactions(id) ON DELETE CASCADE
  fk fk_debts_to_user_id_users: (to_user_id) -> users(id) ON DELETE RESTRICT
  ck ck_debts_amount_positive: (amount_cents > 0)
  ck ck_debts_no_self_debt: (from_user_id <> to_user_id)
  idx ix_debts_from_user_id INDEX: from_user_id
  idx ix_debts_source_transaction_id INDEX: source_transaction_id
  idx ix_debts_to_user_id INDEX: to_user_id

table household
  col base_currency: VARCHAR(3) NOT NULL
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col id: UUID NOT NULL
  col initialized_at: TIMESTAMP WITH TIME ZONE NULL
  col name: VARCHAR(120) NOT NULL
  pk pk_household(id)
  ck ck_household_singleton: (id = '00000000-0000-0000-0000-000000000001'::uuid)

table invitations
  col accepted_at: TIMESTAMP WITH TIME ZONE NULL
  col email: VARCHAR(254) NOT NULL
  col expires_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col id: UUID NOT NULL
  col invited_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col invited_by: UUID NOT NULL
  col revoked_at: TIMESTAMP WITH TIME ZONE NULL
  col token_hash: VARCHAR(64) NOT NULL
  pk pk_invitations(id)
  fk fk_invitations_invited_by_users: (invited_by) -> users(id) ON DELETE RESTRICT
  idx ix_invitations_invited_by INDEX: invited_by
  idx uq_invitations_pending_email UNIQUE: lower((email)::text)) WHERE ((accepted_at IS NULL) AND (revoked_at IS NULL)
  idx uq_invitations_token_hash UNIQUE: token_hash

table refresh_tokens
  col device_label: VARCHAR(120) NULL
  col expires_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col family_id: UUID NOT NULL
  col id: UUID NOT NULL
  col issued_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col parent_id: UUID NULL
  col revoked_at: TIMESTAMP WITH TIME ZONE NULL
  col token_hash: VARCHAR(64) NOT NULL
  col user_id: UUID NOT NULL
  pk pk_refresh_tokens(id)
  fk fk_refresh_tokens_parent_id_refresh_tokens: (parent_id) -> refresh_tokens(id) ON DELETE SET NULL
  fk fk_refresh_tokens_user_id_users: (user_id) -> users(id) ON DELETE CASCADE
  idx ix_refresh_tokens_family_id INDEX: family_id
  idx ix_refresh_tokens_user_id INDEX: user_id
  idx uq_refresh_tokens_token_hash UNIQUE: token_hash

table share_requests
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col id: UUID NOT NULL
  col ratio: NUMERIC(5, 4) NOT NULL
  col requested_by: UUID NOT NULL
  col requested_from: UUID NOT NULL
  col revoked_at: TIMESTAMP WITH TIME ZONE NULL
  col short_label: VARCHAR(100) NOT NULL
  col source_transaction_id: UUID NOT NULL
  pk pk_share_requests(id)
  fk fk_share_requests_requested_by_users: (requested_by) -> users(id) ON DELETE RESTRICT
  fk fk_share_requests_requested_from_users: (requested_from) -> users(id) ON DELETE RESTRICT
  fk fk_share_requests_source_transaction_id_transactions: (source_transaction_id) -> transactions(id) ON DELETE CASCADE
  ck ck_share_requests_no_self: (requested_by <> requested_from)
  idx ix_share_requests_requested_by INDEX: requested_by
  idx ix_share_requests_requested_from INDEX: requested_from
  idx uq_share_requests_active UNIQUE: source_transaction_id, requested_from) WHERE (revoked_at IS NULL

table splits
  col account_id: UUID NOT NULL
  col amount_cents: BIGINT NOT NULL
  col category_id: UUID NULL
  col currency: VARCHAR(3) NOT NULL
  col id: UUID NOT NULL
  col leg_role: VARCHAR NOT NULL
  col savings_goal_id: UUID NULL
  col transaction_id: UUID NOT NULL
  pk pk_splits(id)
  fk fk_splits_account_id_accounts: (account_id) -> accounts(id) ON DELETE RESTRICT
  fk fk_splits_category_id_categories: (category_id) -> categories(id) ON DELETE RESTRICT
  fk fk_splits_transaction_id_transactions: (transaction_id) -> transactions(id) ON DELETE CASCADE
  ck ck_splits_leg_role: (leg_role::text = ANY (ARRAY['funding'::character varying, 'classification'::character varying]::text[]))
  idx ix_splits_account_id INDEX: account_id
  idx ix_splits_category_id INDEX: category_id
  idx ix_splits_transaction_id INDEX: transaction_id

table transactions
  col account_id: UUID NOT NULL
  col category_id: UUID NULL
  col confirmed_at: TIMESTAMP WITH TIME ZONE NULL
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col created_by: UUID NOT NULL
  col date: DATE NOT NULL
  col debt_generation_override: VARCHAR NOT NULL
  col description: VARCHAR NULL
  col id: UUID NOT NULL
  col payee: VARCHAR(255) NULL
  col share_request_id: UUID NULL
  col state: VARCHAR NOT NULL
  col tags: VARCHAR[] NOT NULL
  col voided_at: TIMESTAMP WITH TIME ZONE NULL
  pk pk_transactions(id)
  fk fk_transactions_account_id_accounts: (account_id) -> accounts(id) ON DELETE RESTRICT
  fk fk_transactions_category_id_categories: (category_id) -> categories(id) ON DELETE RESTRICT
  fk fk_transactions_created_by_users: (created_by) -> users(id) ON DELETE RESTRICT
  fk fk_transactions_share_request_id_share_requests: (share_request_id) -> share_requests(id) ON DELETE SET NULL
  ck ck_transactions_debt_generation_override: (debt_generation_override::text = ANY (ARRAY['default'::character varying, 'force_full_debt'::character varying, 'force_no_debt'::character varying]::text[]))
  idx ix_transactions_account_id INDEX: account_id
  idx ix_transactions_category_id INDEX: category_id
  idx ix_transactions_created_by INDEX: created_by

table users
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col disabled_at: TIMESTAMP WITH TIME ZONE NULL
  col display_name: VARCHAR(120) NOT NULL
  col email: VARCHAR(254) NOT NULL
  col id: UUID NOT NULL
  col password_hash: VARCHAR NOT NULL
  col role: user_role NOT NULL
  pk pk_users(id)
  idx uq_users_email_lower UNIQUE: lower((email)::text)
