enum account_type: courant, livret, epargne, especes, credit
enum user_role: admin, member

table account_members
  col account_id: UUID NOT NULL
  col default_share_ratio: NUMERIC(5, 4) NOT NULL
  col id: UUID NOT NULL
  col joined_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col user_id: UUID NOT NULL
  pk pk_account_members(id)
  fk fk_account_members_account_id_accounts: (account_id) -> accounts(id) ON DELETE CASCADE
  fk fk_account_members_user_id_users: (user_id) -> users(id) ON DELETE RESTRICT
  idx ix_account_members_user_id INDEX: user_id
  idx uq_account_members_account_id_user_id UNIQUE: account_id, user_id

table accounts
  col archived_at: TIMESTAMP WITH TIME ZONE NULL
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col currency: VARCHAR(3) NOT NULL
  col household_id: UUID NOT NULL
  col id: UUID NOT NULL
  col name: VARCHAR(120) NOT NULL
  col owner_id: UUID NULL
  col type: account_type NOT NULL
  pk pk_accounts(id)
  fk fk_accounts_household_id_household: (household_id) -> household(id)
  fk fk_accounts_owner_id_users: (owner_id) -> users(id) ON DELETE RESTRICT
  idx ix_accounts_owner_id INDEX: owner_id

table admin_audit_logs
  col action: VARCHAR NOT NULL
  col actor_email: VARCHAR(254) NULL
  col actor_label: VARCHAR(255) NULL
  col actor_user_id: UUID NULL
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col id: UUID NOT NULL
  col metadata: JSONB NULL
  col target_email: VARCHAR(254) NULL
  col target_user_id: UUID NULL
  pk pk_admin_audit_logs(id)
  fk fk_admin_audit_logs_actor_user_id_users: (actor_user_id) -> users(id) ON DELETE SET NULL
  fk fk_admin_audit_logs_target_user_id_users: (target_user_id) -> users(id) ON DELETE SET NULL
  idx ix_admin_audit_logs_actor_user_id_created_at INDEX: actor_user_id, created_at
  idx ix_admin_audit_logs_target_user_id INDEX: target_user_id

table alembic_version
  col version_num: VARCHAR(32) NOT NULL
  pk alembic_version_pkc(version_num)

table household
  col base_currency: VARCHAR(3) NOT NULL
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col id: UUID NOT NULL
  col initialized_at: TIMESTAMP WITH TIME ZONE NULL
  col name: VARCHAR(120) NOT NULL
  pk pk_household(id)
  ck ck_household_singleton: (id = '00000000-0000-0000-0000-000000000001'::uuid)

table invitations
  col accepted_at: TIMESTAMP WITH TIME ZONE NULL
  col email: VARCHAR(254) NOT NULL
  col expires_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col id: UUID NOT NULL
  col invited_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col invited_by: UUID NOT NULL
  col revoked_at: TIMESTAMP WITH TIME ZONE NULL
  col token_hash: VARCHAR(64) NOT NULL
  pk pk_invitations(id)
  fk fk_invitations_invited_by_users: (invited_by) -> users(id) ON DELETE RESTRICT
  idx ix_invitations_invited_by INDEX: invited_by
  idx uq_invitations_pending_email UNIQUE: lower((email)::text)) WHERE ((accepted_at IS NULL) AND (revoked_at IS NULL)
  idx uq_invitations_token_hash UNIQUE: token_hash

table refresh_tokens
  col device_label: VARCHAR(120) NULL
  col expires_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col family_id: UUID NOT NULL
  col id: UUID NOT NULL
  col issued_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col parent_id: UUID NULL
  col revoked_at: TIMESTAMP WITH TIME ZONE NULL
  col token_hash: VARCHAR(64) NOT NULL
  col user_id: UUID NOT NULL
  pk pk_refresh_tokens(id)
  fk fk_refresh_tokens_parent_id_refresh_tokens: (parent_id) -> refresh_tokens(id) ON DELETE SET NULL
  fk fk_refresh_tokens_user_id_users: (user_id) -> users(id) ON DELETE CASCADE
  idx ix_refresh_tokens_family_id INDEX: family_id
  idx ix_refresh_tokens_user_id INDEX: user_id
  idx uq_refresh_tokens_token_hash UNIQUE: token_hash

table users
  col created_at: TIMESTAMP WITH TIME ZONE NOT NULL
  col disabled_at: TIMESTAMP WITH TIME ZONE NULL
  col display_name: VARCHAR(120) NOT NULL
  col email: VARCHAR(254) NOT NULL
  col id: UUID NOT NULL
  col password_hash: VARCHAR NOT NULL
  col role: user_role NOT NULL
  pk pk_users(id)
  idx uq_users_email_lower UNIQUE: lower((email)::text)
